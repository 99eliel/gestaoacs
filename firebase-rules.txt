rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function myUserDoc() {
      return get(/databases/$(database)/documents/usuarios/$(request.auth.uid));
    }

    function isOwnerAdminEmail() {
      return signedIn() && request.auth.token.email in ["carmoeliel99@gmail.com"];
    }

    function isAuthorizedAdminEmail() {
      return signedIn()
        && exists(/databases/$(database)/documents/adminEmails/$(request.auth.token.email))
        && get(/databases/$(database)/documents/adminEmails/$(request.auth.token.email)).data.ativo == true;
    }

    function isAdmin() {
      return signedIn()
        && exists(/databases/$(database)/documents/usuarios/$(request.auth.uid))
        && myUserDoc().data.tipo == "admin";
    }

    match /adminEmails/{emailId} {
      allow create, update, delete: if isAdmin();
      allow read: if isAdmin() || (signedIn() && request.auth.token.email == emailId);
    }

    match /usuarios/{userId} {
      allow create: if signedIn()
        && request.auth.uid == userId
        && request.resource.data.email == request.auth.token.email
        && request.resource.data.nome is string
        && request.resource.data.posto is string
        && request.resource.data.ativo == true
        && request.resource.data.tipo in ["enfermeira", "admin"]
        && (
          request.resource.data.tipo == "enfermeira"
          || (request.resource.data.tipo == "admin" && (isOwnerAdminEmail() || isAuthorizedAdminEmail()))
        );

      allow read: if signedIn() && (request.auth.uid == userId || isAdmin());

      allow update: if signedIn() && (
        isAdmin()
        || (
          request.auth.uid == userId
          && request.resource.data.tipo == resource.data.tipo
          && request.resource.data.email == resource.data.email
        )
      );

      allow delete: if isAdmin();
    }

    match /acs/{acsId} {
      allow create: if signedIn()
        && exists(/databases/$(database)/documents/usuarios/$(request.auth.uid))
        && (
          isAdmin()
          || request.resource.data.enfermeiraId == request.auth.uid
        )
        && request.resource.data.nome is string
        && request.resource.data.posto is string
        && request.resource.data.pessoasCadastradas is number
        && request.resource.data.pessoasCadastradas > 0
        && request.resource.data.ativo == true;

      allow read: if signedIn()
        && (
          isAdmin()
          || resource.data.enfermeiraId == request.auth.uid
        );

      allow update: if signedIn()
        && (
          isAdmin()
          || resource.data.enfermeiraId == request.auth.uid
        );

      allow delete: if isAdmin();
    }

    match /visitas/{visitaId} {
      allow create, update: if signedIn()
        && exists(/databases/$(database)/documents/usuarios/$(request.auth.uid))
        && (
          isAdmin()
          || request.resource.data.enfermeiraId == request.auth.uid
        )
        && request.resource.data.ano is int
        && request.resource.data.mes is int
        && request.resource.data.mes >= 1
        && request.resource.data.mes <= 12
        && request.resource.data.quantidade is number
        && request.resource.data.quantidade >= 0
        && request.resource.data.pessoasCadastradas is number
        && request.resource.data.pessoasCadastradas > 0
        && request.resource.data.coberturaPercentual is number
        && request.resource.data.coberturaPercentual >= 0;

      allow read: if signedIn()
        && (
          isAdmin()
          || resource.data.enfermeiraId == request.auth.uid
        );

      allow delete: if isAdmin();
    }
  }
}
